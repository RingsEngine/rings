# 全局光照

#### Engine3D.setting.gi

配置全局光照(GI)相关参数，基于DDGI(动态漫反射全局光照)探针系统。

**完整属性列表**

| 属性名 | 类型 | 默认值 | 描述 |
|--------|------|--------|------|
| `enable` | boolean | false | 是否启用全局光照 |
| `debug` | boolean | false | 是否启用调试模式 |
| `debugCamera` | boolean | false | 是否启用调试相机视角 |
| **探针体积设置** |
| `probeXCount` | number | 16 | X轴探针数量 |
| `probeYCount` | number | 8 | Y轴探针数量 |
| `probeZCount` | number | 16 | Z轴探针数量 |
| `probeSpace` | number | 64 | 探针间距(单位：世界空间距离) |
| `probeSize` | number | 1.0 | 探针采样数据大小 |
| `offsetX` | number | 0 | 体积X轴偏移 |
| `offsetY` | number | 0 | 体积Y轴偏移 |
| `offsetZ` | number | 0 | 体积Z轴偏移 |
| **渲染质量设置** |
| `probeSourceTextureSize` | number | 128 | 探针纹理分辨率 |
| `octRTMaxSize` | number | 1024 | 八面体纹理总大小 |
| `octRTSideSize` | number | 8 | 单个八面体纹理边长 |
| `maxDistance` | number | 100 | 探针最大深度距离 |
| `rayNumber` | number | 256 | 每探针光线数量 |
| **光照参数** |
| `indirectIntensity` | number | 1.0 | 间接光强度 |
| `bounceIntensity` | number | 1.0 | 反弹光强度 |
| `ddgiGamma` | number | 2.2 | DDGI伽马校正 |
| **实时更新** |
| `realTimeGI` | boolean | false | 是否启用实时全局光照更新 |
| `autoRenderProbe` | boolean | true | 是否自动渲染探针 |
| **高级参数** |
| `normalBias` | number | 0.5 | 法线偏移 |
| `depthSharpness` | number | 100 | 深度锐度 |
| `hysteresis` | number | 0.01 | 滞后值 |
| `lerpHysteresis` | number | 0.01 | 插值滞后值 |
| `irradianceChebyshevBias` | number | 0.5 | 辐照度切比雪夫偏移 |
| `irradianceDistanceBias` | number | 0.1 | 辐照度距离偏移 |
| `probeRoughness` | number | 0.5 | 探针粗糙度 |

**使用示例**

```typescript
import { Engine3D } from '@rings/core';

// 基础配置
Engine3D.setting.gi.enable = true;
Engine3D.setting.gi.indirectIntensity = 1.5;

// 探针体积配置
Engine3D.setting.gi.probeXCount = 32;
Engine3D.setting.gi.probeYCount = 16;
Engine3D.setting.gi.probeZCount = 32;
Engine3D.setting.gi.probeSpace = 64;

// 质量设置
Engine3D.setting.gi.rayNumber = 512;
Engine3D.setting.gi.probeSourceTextureSize = 256;

// 实时更新
Engine3D.setting.gi.realTimeGI = true;
Engine3D.setting.gi.autoRenderProbe = true;
```

    import {
      Engine3D,
      Scene3D,
      Object3D,
      Camera3D,
      View3D,
      Color,
      LitMaterial,
      BoxGeometry,
      PlaneGeometry,
      MeshRenderer,
      AtmosphericComponent,
      DirectLight,
      HoverCameraController,
      GlobalBindGroup,
    } from 'https://unpkg.com/@rings-webgpu/core@1.0.52/dist/rings.es.js';

    async function main() {
      const canvas = document.getElementById('canvas');
      const hud = document.getElementById('hud');
      if (!canvas) return;
      try {
        await Engine3D.init({ canvasConfig: { canvas } });
      } catch (e) {
        document.body.innerHTML =
          '&lt;p style="padding:1rem;font-family:sans-serif"&gt;WebGPU 初始化失败，请使用 Chrome/Edge 并在 localhost 或 HTTPS 下打开。&lt;/p&gt;';
        console.error(e);
        return;
      }

      const scene = new Scene3D();
      scene.addComponent(AtmosphericComponent);

      const camObj = new Object3D();
      const camera = camObj.addComponent(Camera3D);
      camera.perspective(55, canvas.clientWidth / Math.max(canvas.clientHeight, 1), 0.1, 120);
      const hover = camObj.addComponent(HoverCameraController);
      hover.setCamera(30, -28, -11);
      scene.addChild(camObj);

      const sun = new Object3D();
      const dl = sun.addComponent(DirectLight);
      dl.lightColor = new Color(1, 0.95, 0.9, 1);
      dl.intensity = 1.0;
      dl.castShadow = true;
      sun.transform.rotationX = -42;
      sun.transform.rotationY = 25;
      scene.addChild(sun);

      const ground = new Object3D();
      const gr = ground.addComponent(MeshRenderer);
      gr.geometry = new PlaneGeometry(14, 14);
      gr.material = new LitMaterial();
      gr.material.baseColor = new Color(0.25, 0.27, 0.3, 1);
      gr.receiveShadow = true;
      ground.transform.rotationX = -90;
      scene.addChild(ground);

      const boxObj = new Object3D();
      const mr = boxObj.addComponent(MeshRenderer);
      mr.geometry = new BoxGeometry(1, 1, 1);
      mr.material = new LitMaterial();
      mr.material.baseColor = new Color(0.55, 0.6, 0.75, 1);
      mr.castShadow = true;
      boxObj.y = 0.5;
      scene.addChild(boxObj);

      const gien = document.getElementById('gien');
      const ii = document.getElementById('ii');
      const iiV = document.getElementById('ii-v');
      const ps = document.getElementById('ps');
      const psV = document.getElementById('ps-v');
      const rn = document.getElementById('rn');
      const rnV = document.getElementById('rn-v');

      function applyGi() {
        const G = Engine3D.setting.gi;
        G.enable = gien.checked;
        G.indirectIntensity = Number(ii.value) / 100;
        G.probeSpace = Number(ps.value);
        G.rayNumber = Number(rn.value);
        iiV.textContent = G.indirectIntensity.toFixed(2);
        psV.textContent = String(G.probeSpace);
        rnV.textContent = String(G.rayNumber);
        // 否则仅改对象字段不会重填 uniform，看起来「只有初始化有用」
        GlobalBindGroup.getLightEntries(scene).irradianceVolume.setVolumeDataChange();
      }
      applyGi();

      const view = new View3D();
      view.scene = scene;
      view.camera = camera;
      Engine3D.startRenderView(view);

      gien.addEventListener('change', applyGi);
      ii.addEventListener('input', applyGi);
      ps.addEventListener('input', applyGi);
      rn.addEventListener('input', applyGi);

      window.addEventListener('resize', () =&gt; {
        camera.perspective(55, canvas.clientWidth / Math.max(canvas.clientHeight, 1), 0.1, 120);
      });

      function tick() {
        const G = Engine3D.setting.gi;
        hud.textContent = `gi.enable=${G.enable} indirectIntensity=${G.indirectIntensity.toFixed(2)} probeSpace=${G.probeSpace} rayNumber=${G.rayNumber}`;
        requestAnimationFrame(tick);
      }
      requestAnimationFrame(tick);
    }
    main();
  &lt;/script&gt;
