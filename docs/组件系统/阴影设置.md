# 阴影设置

#### Engine3D.setting.shadow

配置阴影相关参数，支持PCF、HARD、SOFT三种阴影类型。

**完整属性列表**

| 属性名 | 类型 | 默认值 | 描述 |
|--------|------|--------|------|
| **基础设置** |
| `enable` | boolean | true | 是否启用阴影 |
| `type` | "PCF" \| "HARD" \| "SOFT" | "HARD" | 阴影类型 |
| **阴影贴图** |
| `shadowSize` | number | 2048 | 方向光阴影贴图分辨率 |
| `pointShadowSize` | number | 2048 | 点光源阴影贴图分辨率 |
| **偏移设置** |
| `shadowBias` | number | 0.05 | 阴影偏移量 |
| `pointShadowBias` | number | 0.005 | 点光源阴影偏移量 |
| **软阴影参数** |
| `shadowSoft` | number | 1.0 | 软阴影强度 |
| **更新控制** |
| `needUpdate` | boolean | true | 是否需要更新阴影 |
| `autoUpdate` | boolean | true | 是否自动更新阴影 |
| `updateFrameRate` | number | 1 | 阴影更新帧率 |
| **级联阴影(CSM)** |
| `csmMargin` | number | 1.0 | CSM边缘扩展 |
| `csmScatteringExp` | number | 1.0 | CSM散射指数 |
| `csmAreaScale` | number | 1.0 | CSM区域缩放 |
| **调试** |
| `debug` | any | null | 调试配置 |
| `shadowBound` | number | - | 阴影边界距离 |

**使用示例**

```typescript
import { Engine3D } from '@rings/core';

// 基础阴影配置
Engine3D.setting.shadow.enable = true;
Engine3D.setting.shadow.type = "PCF";
Engine3D.setting.shadow.shadowSize = 4096;

// 软阴影配置
Engine3D.setting.shadow.shadowSoft = 2.0;

// 性能优化
Engine3D.setting.shadow.autoUpdate = true;
Engine3D.setting.shadow.updateFrameRate = 2;

// 偏移调整
Engine3D.setting.shadow.shadowBias = 0.01;
Engine3D.setting.shadow.pointShadowBias = 0.001;

// CSM配置
Engine3D.setting.shadow.csmMargin = 2.0;
```

#### 可运行示例（Engine3D.setting.shadow）

切换 **`shadow.type`**（HARD / PCF / SOFT），调节 **shadowBias**、**shadowSoft**、**shadowSize**（2 的幂），观察地面与立方体阴影。

    import {
      Engine3D,
      Scene3D,
      Object3D,
      Camera3D,
      View3D,
      Color,
      LitMaterial,
      BoxGeometry,
      PlaneGeometry,
      MeshRenderer,
      AtmosphericComponent,
      DirectLight,
      HoverCameraController,
    } from 'https://unpkg.com/@rings-webgpu/core@1.0.52/dist/rings.es.js';

    async function main() {
      const canvas = document.getElementById('canvas');
      const hud = document.getElementById('hud');
      if (!canvas) return;
      try {
        await Engine3D.init({ canvasConfig: { canvas } });
      } catch (e) {
        document.body.innerHTML =
          '&lt;p style="padding:1rem;font-family:sans-serif"&gt;WebGPU 初始化失败，请使用 Chrome/Edge 并在 localhost 或 HTTPS 下打开。&lt;/p&gt;';
        console.error(e);
        return;
      }

      const scene = new Scene3D();
      scene.addComponent(AtmosphericComponent);

      const camObj = new Object3D();
      const camera = camObj.addComponent(Camera3D);
      camera.perspective(50, canvas.clientWidth / Math.max(canvas.clientHeight, 1), 0.1, 150);
      const hover = camObj.addComponent(HoverCameraController);
      hover.setCamera(35, -25, 14);
      scene.addChild(camObj);

      const sun = new Object3D();
      const dl = sun.addComponent(DirectLight);
      dl.lightColor = new Color(1, 1, 1, 1);
      dl.intensity = 1.2;
      dl.castShadow = true;
      sun.transform.rotationX = -55;
      sun.transform.rotationY = 40;
      scene.addChild(sun);

      const ground = new Object3D();
      const gr = ground.addComponent(MeshRenderer);
      gr.geometry = new PlaneGeometry(20, 20);
      gr.material = new LitMaterial();
      gr.material.baseColor = new Color(0.85, 0.86, 0.88, 1);
      gr.receiveShadow = true;
      ground.transform.rotationX = -90;
      scene.addChild(ground);

      const boxObj = new Object3D();
      const mr = boxObj.addComponent(MeshRenderer);
      mr.geometry = new BoxGeometry(1.2, 2.2, 1.2);
      mr.material = new LitMaterial();
      mr.material.baseColor = new Color(0.55, 0.5, 0.45, 1);
      mr.castShadow = true;
      boxObj.y = 1.1;
      scene.addChild(boxObj);

      const view = new View3D();
      view.scene = scene;
      view.camera = camera;
      Engine3D.startRenderView(view);

      const stype = document.getElementById('stype');
      const sb = document.getElementById('sb');
      const sbV = document.getElementById('sb-v');
      const ss = document.getElementById('ss');
      const ssV = document.getElementById('ss-v');
      const sz = document.getElementById('sz');
      const szV = document.getElementById('sz-v');

      function applyShadow() {
        const S = Engine3D.setting.shadow;
        S.enable = true;
        S.type = stype.value;
        S.shadowBias = Number(sb.value) / 100;
        S.shadowSoft = Number(ss.value) / 100;
        S.shadowSize = 1 &lt;&lt; Number(sz.value);
        sbV.textContent = S.shadowBias.toFixed(2);
        ssV.textContent = S.shadowSoft.toFixed(2);
        szV.textContent = String(S.shadowSize);
      }
      stype.addEventListener('change', applyShadow);
      sb.addEventListener('input', applyShadow);
      ss.addEventListener('input', applyShadow);
      sz.addEventListener('input', applyShadow);
      applyShadow();

      window.addEventListener('resize', () =&gt; {
        camera.perspective(50, canvas.clientWidth / Math.max(canvas.clientHeight, 1), 0.1, 150);
      });

      function tick() {
        const S = Engine3D.setting.shadow;
        hud.textContent = `shadow.type=${S.type} bias=${S.shadowBias.toFixed(2)} soft=${S.shadowSoft.toFixed(2)} size=${S.shadowSize}`;
        requestAnimationFrame(tick);
      }
      requestAnimationFrame(tick);
    }
