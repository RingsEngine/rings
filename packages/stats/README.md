## Usage

```bash
npm install @rings/core --save
npm install @rings/stats --save
```

```ts
import { Scene3D } from "@rings/core";
import { Stats } from "@rings/stats";

let scene = new Scene3D();
scene.addComponent(Stats);
```

```js
const { Scene3D, Stats } = Rings;

let scene = new Scene3D();
scene.addComponent(Stats);
```
